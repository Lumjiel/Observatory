// ChatInput.tsx - 聊天输入框
export function ChatInput() {
  // 状态
  const [value, setValue] = React.useState("");
  const [isFocused, setIsFocused] = React.useState(false);
  
  // 引用
  const textareaRef = React.useRef<HTMLTextAreaElement>(null);
  
  // 自动调整高度
  const adjustHeight = React.useCallback(() => {
    const el = textareaRef.current;
    if (!el) return;
    el.style.height = "auto";
    el.style.height = `${Math.min(el.scrollHeight, 160)}px`;
  }, []);
  
  // 发送消息
  const handleSubmit = async () => {
    if (isStreaming) {
      cancelGeneration();  // 停止生成
      return;
    }
    await sendMessage(value);  // 发送消息
  };
  
  return (
    <div className="flex items-end gap-2">
      <Textarea
        ref={textareaRef}
        value={value}
        onChange={(e) => setValue(e.target.value)}
        onKeyDown={handleKeyDown}
        placeholder="输入问题..."
      />
      <Button onClick={handleSubmit}>
        {isStreaming ? <Square /> : <Send />}
      </Button>
    </div>
  );
}

#### Axios 实例配置

```typescript
// services/api.ts
import axios from "axios";
import { storage } from "@/utils/storage";

export const api = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || "",
  timeout: 60000
});

// 请求拦截器：添加 Token
api.interceptors.request.use((config) => {
  const token = storage.getToken();
  if (token) {
    config.headers.Authorization = token;
  }
  return config;
});

// 响应拦截器：统一错误处理
api.interceptors.response.use(
  (response) => {
    const payload = response.data;
    // 业务错误处理
    if (payload.code !== "0") {
      const message = payload.message || "请求失败";
      // Token 过期，跳转登录
      if (message.includes("未登录")) {
        storage.clearAuth();
        window.location.href = "/login";
      }
      return Promise.reject(new Error(message));
    }
    return payload.data;
  },
  (error) => {
    // HTTP 错误处理
    if (error?.response?.status === 401) {
      storage.clearAuth();
      window.location.href = "/login";
    }
    return Promise.reject(error);
  }
);
```

export const useChatStore = create<ChatState>((set, get) => ({

  currentSessionId: null,

  isStreaming: false,

        question: content,
        conversationId: get().currentSessionId

      // 处理响应
      // ...
    } finally {
      set({ isLoading: false });
    }
  }
}));

#### 流式响应 Hook

```typescript
// hooks/useStreamResponse.ts
export interface StreamHandlers {
  onMeta?: (payload: StreamMetaPayload) => void;
  onMessage?: (payload: MessageDeltaPayload) => void;
  onThinking?: (payload: MessageDeltaPayload) => void;
  onFinish?: (payload: CompletionPayload) => void;
  onError?: (error: Error) => void;
}

export async function readSseStream(
  response: Response,
  handlers: StreamHandlers,
  signal?: AbortSignal
) {
  const reader = response.body.getReader();
  const decoder = new TextDecoder("utf-8");
  let buffer = "";
  
  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    
    buffer += decoder.decode(value, { stream: true });
    
    // 按行解析 SSE 事件
    const lines = buffer.split("\n");
    buffer = lines.pop() || "";
    
    for (const line of lines) {
      if (line.startsWith("event:")) {
        // 事件类型
      } else if (line.startsWith("data:")) {
        // 数据内容
        const data = JSON.parse(line.slice(5));
        handlers.onMessage?.(data);
      }
    }
  }
}
```

## 💡 八、开发建议

### 8.1 新增页面步骤

```
1. 在 src/pages/admin/ 下创建页面组件
2. 在 router.tsx 中添加路由
3. 在 AdminLayout.tsx 中添加菜单项
4. 在对应的 services/ 中添加 API 方法
```

### 8.2 新增组件步骤

```
1. UI 组件：放在 src/components/ui/
2. 业务组件：放在对应的功能目录下
3. 使用 shadcn/ui：运行 npx shadcn@latest add button
```

### 8.3 状态管理原则

```
✅ 页面级状态：使用 React useState
✅ 跨组件状态：使用 Zustand Store
✅ 服务端状态：使用 API Service + Store
```

这个前端架构设计清晰，模块化程度高，便于维护和扩展！
